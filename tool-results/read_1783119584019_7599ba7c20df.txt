   326→function velure3_get_image_url( $img, $fallback = '' ) {
   327→        if ( empty( $img ) ) return $fallback;
   328→        if ( is_array( $img ) ) return isset( $img['url'] ) ? $img['url'] : $fallback;
   329→        if ( is_numeric( $img ) ) return wp_get_attachment_url( $img ) ?: $fallback;
   330→        return esc_url( $img );
   331→}
   332→
   333→function velure3_opt( $key, $fallback = '' ) {
   334→        if ( ! function_exists( 'get_field' ) ) return $fallback;
   335→        $val = get_field( $key, 'option' );
   336→        return ( $val !== null && $val !== '' && $val !== array() ) ? $val : $fallback;
   337→}
   338→
   339→function velure3_opt_bool( $key, $fallback = true ) {
   340→        if ( ! function_exists( 'get_field' ) ) return $fallback;
   341→        $val = get_field( $key, 'option' );
   342→        return ( $val !== null && $val !== '' ) ? (bool) $val : $fallback;
   343→}
