    16→/**
    17→ * Recuperer l'URL d'une image (ID entier ou URL brute).
    18→ */
    19→function velure_core_get_image_url( $img, $fallback = '' ) {
    20→        if ( empty( $img ) ) return $fallback;
    21→        if ( is_array( $img ) ) return isset( $img['url'] ) ? $img['url'] : $fallback;
    22→        if ( is_numeric( $img ) ) return wp_get_attachment_url( (int) $img ) ?: $fallback;
    23→        return esc_url( $img );
    24→}

    26→/**
    27→ * Lire un parametre Velure Core avec fallback.
    28→ * Lit depuis l'option wp velure_core_settings.
    29→ */
    30→function velure_core_opt( $key, $fallback = '' ) {
    31→        static $settings = null;
    32→        if ( null === $settings ) {
    33→                $settings = get_option( VELURE_CORE_OPTION, array() );
    34→                if ( ! is_array( $settings ) ) $settings = array();
    35→        }
    36→        $val = isset( $settings[ $key ] ) ? $settings[ $key ] : null;
    37→        if ( $val === null || $val === '' ) return $fallback;
    38→        if ( is_array( $val ) && empty( $val ) ) return $fallback;
    39→        return $val;
    40→}
    41→
    42→/**
    43→ * Lire un bool Velure Core avec fallback.
    44→ */
    45→function velure_core_opt_bool( $key, $fallback = true ) {
    46→        static $settings = null;
    47→        if ( null === $settings ) {
    48→                $settings = get_option( VELURE_CORE_OPTION, array() );
    49→                if ( ! is_array( $settings ) ) $settings = array();
    50→        }
    51→        if ( ! isset( $settings[ $key ] ) ) return $fallback;
    52→        return (bool) $settings[ $key ];
    53→}

   110→function velure_core_get_ordered_sections() {
   111→        $default_order = array( 'hero', 'features', 'categories', 'products', 'split_banner', 'marquee', 'testimonials', 'blog', 'instagram' );
   112→        $custom_order  = velure_core_opt( 'section_order', array() );
   113→        if ( ! empty( $custom_order ) && is_array( $custom_order ) ) {
   114→                $order = array_values( $custom_order );
   115→        } else {
   116→                $order = $default_order;
   117→        }
   118→        $visibility = velure_core_get_section_visibility();
   119→        return array_filter( $order, function( $sec ) use ( $visibility ) {
   120→                return isset( $visibility[ $sec ] ) && $visibility[ $sec ];
   121→        } );
   122→}

   377→/* ── 4I. INSTAGRAM ── */
   378→function velure_core_get_instagram_items() {
   379→        $uri   = velure_core_theme_uri();
   380→        $items = velure_core_opt( 'instagram_images', array() );
   381→        if ( ! empty( $items ) && is_array( $items ) ) {
   382→                $ig_url = velure_core_opt( 'instagram_url', 'https://instagram.com/' );
   383→                $out = array();
   384→                foreach ( $items as $item ) {
   385→                        if ( empty( $item['image'] ) ) continue;
   386→                        $out[] = array( 'image' => velure_core_get_image_url( $item['image'] ), 'link' => ! empty( $item['link'] ) ? $item['link'] : $ig_url, 'alt' => $item['alt'] ?? '' );
   387→                }
   388→                if ( ! empty( $out ) ) return $out;
   389→        }
   390→        $defaults = array();
   391→        $ig_url = velure_core_opt( 'instagram_url', 'https://instagram.com/' );
   392→        for ( $i = 1; $i <= 6; $i++ ) {
   393→                $defaults[] = array( 'image' => $uri . '/assets/images/instagram-0' . $i . '.jpg', 'link' => $ig_url, 'alt' => '' );
   394→        }
   395→        return $defaults;
   396→}

   400→   5. FRONT-END INJECTION (custom CSS, etc.)
   401→   ═══════════════════════════════════════════════ */
   402→add_action( 'wp_head', 'velure_core_inject_custom_css' );
   403→function velure_core_inject_custom_css() {
   404→        if ( ! is_front_page() ) return;
   405→
   406→        $rules = array();
   407→
   408→        /* Section padding */
   409→        $pad = velure_core_opt( 'section_padding', 'normal' );
   410→        $pad_map = array( 'compact' => '3rem', 'normal' => '5.5rem', 'spacious' => '8rem', 'none' => '0' );
   411→        if ( isset( $pad_map[ $pad ] ) ) {
   412→                $rules[] = '.velure-section { padding-top: ' . $pad_map[ $pad ] . '; padding-bottom: ' . $pad_map[ $pad ] . '; }';
   413→        }
   414→
   415→        /* Custom CSS from admin */
   416→        $custom_css = velure_core_opt( 'custom_css', '' );
   417→        if ( $custom_css ) {
   418→                $rules[] = $custom_css;
   419→        }
   420→
   421→        if ( empty( $rules ) ) return;
   422→
   423→        echo '<style id="velure-core-custom">' . "\n";
   424→        echo implode( "\n", $rules );
   425→        echo "\n</style>\n";
   426→}
