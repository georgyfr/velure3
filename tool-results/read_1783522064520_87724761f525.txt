    69→        /* ═══════════════════════════════════════
    70→           3. REPEATER — Add / Remove / Sort
    71→           ═══════════════════════════════════════ */
    72→        _addRepeaterRow: function(templateEl, containerEl, label) {
    73→                var clone = templateEl.clone().removeClass('vc-repeater-template').removeAttr('id').show();
    74→                var idx = containerEl.find('.vc-repeater-row:not(.vc-repeater-template)').length;
    75→
    76→                clone.find('[name]').each(function(){
    77→                        var name = $(this).attr('name');
    78→                        if (!name) return;
    79→                        name = name.replace(/\[\d+\]/g, '[' + idx + ']');
    80→                        name = name.replace(/__INDEX__/g, idx);
    81→                        name = name.replace(/__NUM__/g, (idx + 1));
    82→                        $(this).attr('name', name);
    83→                });
    84→                clone.find('[data-index]').attr('data-index', idx);
    85→
    86→                clone.find('.vc-repeater-title').text((label || 'Element') + ' ' + (idx + 1));
    87→                clone.addClass('open');
    88→                containerEl.append(clone);
    89→        },

   260→        initTemplates: function() {
   261→                var self = this;
   262→                $(document).on('click', '.vc-template-card', function(e){
   263→                        e.preventDefault();

   265→                        if (!confirm('Appliquer ce modele ? Les reglages actuels de cette section seront remplaces.')) return;

   269→                        var templates = {};
   270→                        try {
   271→                                templates = JSON.parse($('#vc-templates-data').text() || '{}');
   272→                        } catch(ex) {
   273→                                self.toast('Erreur de lecture des modeles.', 'error');
   274→                                return;

   278→                        var tmpl = null;
   279→                        for (var i = 0; i < sectionTemplates.length; i++) {
   280→                                if (sectionTemplates[i].name === name) { tmpl = sectionTemplates[i]; break; }
   281→                        }
   282→                        if (!tmpl || !tmpl.data) {
   283→                                self.toast('Modele introuvable.', 'error');
   284→                                return;
   285→                        }
   286→
   287→                        /* Fill form fields from template data */
   288→                        var skipKeys = ['hero_slides','trust_features','brand_names','instagram_images'];
   289→                        $.each(tmpl.data, function(key, val){
   290→                                if (skipKeys.indexOf(key) !== -1) return;
   291→                                var field = $('[name="' + key + '"]');
   292→                                if (field.length) {
   293→                                        if (field.is(':checkbox')) {
   294→                                                field.prop('checked', !!val);
   295→                                        } else {
   296→                                                field.val(val).trigger('change');
   297→                                        }
   298→                                }
   299→                        });
   300→
   301→                        /* Trigger conditional fields */
   302→                        $('[name="hero_show_side"]').trigger('change');
   303→
   304→                        self.markUnsaved();
   305→                        self.toast('Modele "' + name + '" applique !', 'success');
   306→                });
   307→        },

   321→        /* ═══════════════════════════════════════
   322→           11. AJAX PUBLISH
   323→           ═══════════════════════════════════════ */
   324→        initPublish: function() {
   325→                var self = this;
   326→
   327→                /* Re-find the form every time (more reliable than caching) */
   328→                $('#vc-publish-btn').on('click', function(e){
   329→                        e.preventDefault();
   330→
   331→                        var $form = $('#vc-settings-form, #vc-builder-form').first();
   332→                        if (!$form.length) {
   333→                                self.toast('Formulaire introuvable. Rechargez la page.', 'error');
   334→                                console.error('[Velure Core] #vc-settings-form not found');
   335→                                return;
   336→                        }
   337→
   338→                        var btn = $(this);
   339→                        btn.prop('disabled', true).html('<span style="display:inline-flex;align-items:center;gap:6px"><span class="vc-spinner"></span> Publication...</span>');
   340→                        self.toast('Publication en cours...', 'saving');
   341→
   342→                        /* Build data object explicitly for reliability */
   343→                        var formData = $form.serialize();
   344→                        var ajaxUrl = (typeof velureCoreAdmin !== 'undefined') ? velureCoreAdmin.ajaxUrl : '/wp-admin/admin-ajax.php';
   345→
   346→                        $.ajax({
   347→                                url: ajaxUrl,
   348→                                type: 'POST',
   349→                                data: formData,
   350→                                dataType: 'json',
   351→                                success: function(res){
   352→                                        if (res && res.success) {
   353→                                                self.markSaved();
   354→                                                self.toast(res.data && res.data.message ? res.data.message : 'Publie !', 'success');
   355→                                                $('.vc-footer-status span:last-child').text('Derniere sauvegarde : ' + new Date().toLocaleTimeString());
   356→                                                VC._initialForm = $form.serialize();
   357→                                        } else {
   358→                                                var errMsg = (res && res.data && res.data.message) ? res.data.message : 'Erreur lors de la publication.';
   359→                                                self.toast(errMsg, 'error');
   360→                                                console.error('[Velure Core] Save error:', res);
   361→                                        }
   362→                                },
   363→                                error: function(xhr, status, err){
   364→                                        self.toast('Erreur reseau (' + status + '). Veuillez reessayer.', 'error');
   365→                                        console.error('[Velure Core] AJAX error:', status, err);
   366→                                },
   367→                                complete: function(){
   368→                                        btn.prop('disabled', false).html('&#10003; Publier');
   369→                                }
   370→                        });
   371→                });
   372→
   373→                /* Discard */
   374→                $('#vc-discard-btn').on('click', function(){
   375→                        if (VC.unsaved && !confirm('Voulez-vous vraiment annuler toutes les modifications non publiees ?')) return;
   376→                        window.location.reload();
   377→                });
   378→        },

   383→        initIO: function() {

   385→                var ajaxUrl = (typeof velureCoreAdmin !== 'undefined') ? velureCoreAdmin.ajaxUrl : '/wp-admin/admin-ajax.php';

   404→                        var ta = document.getElementById('vc-export-text');

   406→                                ta.select();

   408→                        }
   409→                        var btn = $(this);
   410→                        btn.text('\u2713 Copie !');
   411→                        setTimeout(function(){ btn.text('&#128203; Copier'); }, 2000);

   425→                                JSON.parse(json);

   427→                                self.toast('JSON invalide.', 'error');
   428→                                return;
   429→                        }
   430→                        $.post(ajaxUrl, {
   431→                                action: 'velure_core_import_settings',
   432→                                _wpnonce: nonce,
   433→                                settings: json
   434→                        }, function(res){
   435→                                if (res && res.success) {
   436→                                        self.toast('Import reussi ! Rechargement...', 'success');
   437→                                        setTimeout(function(){ window.location.reload(); }, 1500);
   438→                                } else {
   439→                                        self.toast((res && res.data && res.data.message) ? res.data.message : "Erreur d'import.", 'error');
   440→                                }
   441→                        });
   442→                });

   448→                                action: 'velure_core_reset',
   449→                                _wpnonce: nonce
   450→                        }, function(res){
   451→                                if (res && res.success) {
   452→                                        self.toast('Reinitialisation reussie. Rechargement...', 'success');
   453→                                        setTimeout(function(){ window.location.reload(); }, 1500);
   454→                                }
   455→                        });
   456→                });
   457→        },

   461→           ═══════════════════════════════════════ */
   462→        toast: function(msg, type) {
   463→                var el = $('#vc-toast');
   464→                if (!el.length) {
   465→                        /* Fallback: create toast if missing */
   466→                        $('body').append('<div id="vc-toast" class="vc-toast"></div>');
   467→                        el = $('#vc-toast');
   468→                }
   469→                var icon = '';
   470→                if (type === 'success') icon = '\u2713 ';
   471→                else if (type === 'error') icon = '\u2717 ';
   472→                else if (type === 'saving') icon = '\u23F3 ';
   473→
   474→                el.removeClass('show vc-toast-success vc-toast-error vc-toast-saving');
   475→                if (type) el.addClass('vc-toast-' + type);
   476→                el.html(icon + msg).addClass('show');
   477→                clearTimeout(VC._toastTimer);
   478→                VC._toastTimer = setTimeout(function(){ el.removeClass('show'); }, 4000);
   479→        },
