   696→                onFieldChange: function (key, value) {
   697→                        if (!VCB.selected) return;

   699→
   700→                        /* ── Queue change for auto-save (debounce 1.5s) ── */

   708→                                var cssVal = isNaN(numVal) ? value : numVal;
   709→                                if (cssDef.unit) cssVal = cssVal + cssDef.unit;

   716→                                        css[cssDef.also] = alsoVal;

   719→                                VCB.sendToIframe({ action: 'vc-apply-style', widget: widgetId, css: css });
   720→                                return;
   721→                        }

   724→                        var base = VCB.selected.base || widgetId;
   725→                        if (base === 'hero-eyebrow' || base === 'hero-title' || base === 'hero-subtitle') {
   726→                                VCB.sendToIframe({ action: 'vc-apply-text', widget: widgetId, text: value });
   727→                        }
   728→
   729→                        /* For hero-cta text */
   730→                        if (base === 'hero-cta' && key.indexOf('cta_text') !== -1) {
   731→                                VCB.sendToIframe({ action: 'vc-apply-text', widget: widgetId, text: value });
   732→                        }

   788→                                VCB.saveChanges();

   792→                /**
   793→                 * Flush all pending changes to the server via AJAX.
   794→                 * Called by the debounce timer or by the Publish button.
   795→                 */
   796→                saveChanges: function () {
   797→                        var keys = Object.keys(VCB.pendingChanges);
   798→                        if (keys.length === 0) {
   799→                                VCB._showSaveIndicator('idle');
   800→                                return;
   801→                        }
   802→
   803→                        var changes = JSON.parse(JSON.stringify(VCB.pendingChanges));
   804→                        VCB._showSaveIndicator('saving');
   805→
   806→                        $.ajax({
   807→                                url:  (typeof velureCoreAdmin !== 'undefined') ? velureCoreAdmin.ajaxUrl : '/wp-admin/admin-ajax.php',
   808→                                type: 'POST',
   809→                                data: {
   810→                                        action:  'velure_core_auto_save',
   811→                                        _wpnonce: (typeof velureCoreAdmin !== 'undefined') ? velureCoreAdmin.nonce : '',
   812→                                        changes: JSON.stringify(changes),
   813→                                },
   814→                                success: function (res) {
   815→                                        if (res && res.success) {
   816→                                                keys.forEach(function (k) {
   817→                                                        VCB._setNestedValue(VCB.settings, k, changes[k]);
   818→                                                });
   819→                                                VCB._markSaved();
   820→                                                VCB._showSaveIndicator('saved');
   821→                                        } else {
   822→                                                VCB._showSaveIndicator('error');
   823→                                                VCB.toast((res && res.data && res.data.message) || 'Erreur de sauvegarde.', 'error');
   824→                                        }
   825→                                },
   826→                                error: function () {
   827→                                        VCB._showSaveIndicator('error');
   828→                                        VCB.toast('Erreur reseau lors de la sauvegarde.', 'error');
   829→                                },
   830→                        });
   831→                },

   833→                _setNestedValue: function (obj, key, value) {
   834→                        var parts = key.split('.');
   835→                        var current = obj;
   836→                        for (var i = 0; i < parts.length - 1; i++) {
   837→                                if (!current[parts[i]] || typeof current[parts[i]] !== 'object') {
   838→                                        current[parts[i]] = {};
   839→                                }
   840→                                current = current[parts[i]];
   841→                        }
   842→                        current[parts[parts.length - 1]] = value;
   843→                },
