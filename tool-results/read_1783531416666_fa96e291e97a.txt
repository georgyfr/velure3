   842→   842→function velure_core_ajax_section_form() {
   843→   843→        check_ajax_referer( 'velure_core_save_settings' );
   844→   844→        if ( ! current_user_can( 'edit_theme_options' ) ) wp_send_json_error( array( 'message' => 'Permission insuffisante.' ) );
   845→   845→
   846→   846→        $section = isset( $_POST['section'] ) ? sanitize_text_field( $_POST['section'] ) : '';
   847→   847→        $valid   = velure_core_canvas_valid_sections();
   848→   848→        if ( ! isset( $valid[ $section ] ) ) wp_send_json_error( array( 'message' => 'Section invalide.' ) );

   851→   851→                'hero'         => 'velure_core_tab_hero',

   860→   860→        );
