    32→/**
    33→ * Check if the current request is a canvas render request.
    34→ * Returns the section slug if valid, false otherwise.
    35→ */
    36→function velure_core_is_canvas_request() {
    37→        if ( ! is_admin() ) {
    38→                return false;
    39→        }
    40→        if ( ! isset( $_GET['mode'] ) || 'canvas' !== $_GET['mode'] ) {
    41→                return false;
    42→        }
    43→        if ( ! current_user_can( 'edit_theme_options' ) ) {
    44→                return false;
    45→        }
    46→        $section = isset( $_GET['section'] ) ? sanitize_text_field( $_GET['section'] ) : '';
    47→        $valid   = velure_core_canvas_valid_sections();
    48→        if ( ! isset( $valid[ $section ] ) ) {
    49→                return false;
    50→        }
    51→        return $section;
    52→}
